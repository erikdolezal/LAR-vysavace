  7class Objects:
  8    """
  9    A class to represent objects in a simulation.
 10    Attributes:
 11        sim (object): The simulation instance the object belongs to.
 12        type (str, optional): The type of the object. Defaults to None.
 13        RADIUS (int): The radius of the object. Defaults to 0.
 14        pos (numpy.ndarray): The position of the object as a 2D array. Initialized if x and y are provided.
 15    Methods:
 16        DimToPixels(dim_real):
 17            Converts a real-world dimension to pixels based on simulation parameters.
 18        was_collision(collission_pos, radius):
 19            Checks if a collision occurred with another object based on its position and radius.
 20        get_position():
 21            Returns the current position of the object.
 22    """
 23
 24    def __init__(self, sim, type=None, x=None, y=None):
 25
 26        # object parametrs
 27        self.sim = sim
 28        self.type = type
 29        self.RADIUS = 0
 30        if x is not None and y is not None:
 31            self.pos = np.array([x, y])
 32
 33    def DimToPixels(self, dim_real):
 34        """
 35        Converts a real-world dimension to its equivalent in pixels.
 36        Args:
 37            dim_real (float): The real-world dimension to be converted.
 38        Returns:
 39            int: The dimension in pixels, calculated based on the ratio of
 40                 simulation width in pixels to the real-world side length.
 41        """
 42        return int((SimParm.WIDTH / SimParm.SIDE_REAL) * dim_real)
 43
 44    def was_collision(self, collission_pos, radius):
 45        """
 46        Determines whether a collision has occurred between the object and another object.
 47        Args:
 48            collission_pos (numpy.ndarray): The position of the other object as a NumPy array.
 49            radius (float): The radius of the other object.
 50        Returns:
 51            bool: True if a collision has occurred, False otherwise.
 52        """
 53        distance = np.linalg.norm(collission_pos - self.pos)
 54        if distance <= self.RADIUS + radius:
 55            return True
 56        else:
 57            return False
 58
 59    def get_position(self):
 60        """
 61        Retrieve the current position of the object.
 62        Returns:
 63            tuple: The current position of the object as a tuple (e.g., (x, y)).
 64        """
 65
 66        return self.pos

116        RADIUS (float): The radius of the TurtleBot, defined in simulation parameters.
117    Methods:
118        __init__(sim, type, x, y, angle):
119            Initializes a TurtleBot object with its simulation context, type, position, and angle.
120        draw():
121            Draws the TurtleBot on the simulation screen, including its position, orientation, and field of view.
122        move(velocity, angular_velocity, forward=True, clockwise=True):
123            Moves the TurtleBot based on the given velocity and angular velocity.
124            The direction of movement can be controlled using the `forward` and `clockwise` flags.
125        get_info():
126            Returns the current state of the TurtleBot as a NumPy array, including position, angle, and last velocity.
127    """
128
129    def __init__(self, sim, type, x, y, angle):
130        super().__init__(sim, type, x, y)
131        # turlte parametrs
132        self.angle = angle
133        self.last_velosity = 0
134        # turtle constants
135        self.RADIUS = SimParm.TURTLE_RADIUS
136
137    def draw(self):
138        """
139        Draws the TurtleBot on the simulation screen.
140        This method visualizes the TurtleBot's position, orientation, and field of view (FOV)
141        on the simulation screen using Pygame. It includes:
142        - A circle representing the TurtleBot's body.
143        - A line indicating the front direction of the TurtleBot.
144        - Two lines representing the edges of the TurtleBot's camera field of view.
145        The drawing is scaled and positioned based on the TurtleBot's current position,
146        orientation, and dimensions.
147        Attributes:
148            self.pos (tuple): The (x, y) position of the TurtleBot in the simulation.
149            self.angle (float): The orientation angle of the TurtleBot in radians.
150            self.RADIUS (float): The radius of the TurtleBot.
151            self.sim.screen (pygame.Surface): The Pygame surface where the TurtleBot is drawn.
152            SimParm.RED (tuple): The RGB color for the TurtleBot's body.
153            SimParm.BLACK (tuple): The RGB color for the lines.
154            SimParm.CAMERA_FOV (float): The field of view angle of the TurtleBot's camera in radians.
155            SimParm.FOV_LINE_LENGTH (float): The length of the FOV lines.
156        Note:
157            The `DimToPixels` method is used to convert dimensions from simulation units
158            to pixel units for rendering on the screen.
159        """
160
161        # Robot's front direction (for simulation of movement)
162        line_length = self.DimToPixels(self.RADIUS) * 2
163        line_end_x = self.DimToPixels(self.pos[0]) + np.cos(self.angle) * line_length
164        line_end_y = self.DimToPixels(self.pos[1]) - np.sin(self.angle) * line_length
165
166        fov_line_x1 = self.DimToPixels(self.pos[0]) + np.cos(self.angle + SimParm.CAMERA_FOV / 2) * SimParm.FOV_LINE_LENGTH
167        fov_line_y1 = self.DimToPixels(self.pos[1]) - np.sin(self.angle + SimParm.CAMERA_FOV / 2) * SimParm.FOV_LINE_LENGTH
168        fov_line_x2 = self.DimToPixels(self.pos[0]) + np.cos(self.angle - SimParm.CAMERA_FOV / 2) * SimParm.FOV_LINE_LENGTH
169        fov_line_y2 = self.DimToPixels(self.pos[1]) - np.sin(self.angle - SimParm.CAMERA_FOV / 2) * SimParm.FOV_LINE_LENGTH
170
171        pygame.draw.circle(self.sim.screen, SimParm.RED, (self.DimToPixels(self.pos[0]), self.DimToPixels(self.pos[1])), self.DimToPixels(self.RADIUS))
172        pygame.draw.line(self.sim.screen, SimParm.BLACK, (self.DimToPixels(self.pos[0]), self.DimToPixels(self.pos[1])), (line_end_x, line_end_y), 3)
173        pygame.draw.line(self.sim.screen, SimParm.BLACK, (self.DimToPixels(self.pos[0]), self.DimToPixels(self.pos[1])), (fov_line_x1, fov_line_y1), 4)
174        pygame.draw.line(self.sim.screen, SimParm.BLACK, (self.DimToPixels(self.pos[0]), self.DimToPixels(self.pos[1])), (fov_line_x2, fov_line_y2), 4)
175
176    def move(self, velocity, angular_velocity, forward=True, clockwise=True):
177        """
178        Moves the robot based on the specified velocity and angular velocity.
179        Parameters:
180            velocity (float): The linear velocity of the robot.
181            angular_velocity (float): The angular velocity of the robot.
182            forward (bool, optional): If True, the robot moves forward. If False, the robot moves backward. Defaults to True.
183            clockwise (bool, optional): If True, the robot rotates clockwise. If False, the robot rotates counterclockwise. Defaults to True.
184        Updates:
185            - The robot's angle is adjusted based on the angular velocity and direction.
186            - The robot's position is updated based on the linear velocity, direction, and current angle.
187            - The `last_velocity` attribute is set to the provided velocity.
188        """
189
190        if clockwise:
191            self.angle += angular_velocity / SimParm.SIM_FPS
192        else:
193            self.angle -= angular_velocity / SimParm.SIM_FPS
194        self.angle = (self.angle + np.pi) % (2 * np.pi) - np.pi
195
196        if forward:
197            self.pos[0] += velocity / SimParm.SIM_FPS * np.cos(self.angle)
198            self.pos[1] -= velocity / SimParm.SIM_FPS * np.sin(self.angle)
199        else:
200            self.pos[0] -= velocity / SimParm.SIM_FPS * np.cos(self.angle)
201            self.pos[1] += velocity / SimParm.SIM_FPS * np.sin(self.angle)
202
203        self.last_velosity = velocity
204
205    def get_info(self):
206        """
207        Retrieve information about the object's current state.
208        Returns:
209            numpy.ndarray: A 1D array containing the following elements:
210                - self.pos[0] (float): The x-coordinate of the object's position.
211                - self.pos[1] (float): The y-coordinate of the object's position.
212                - self.angle (float): The object's current angle or orientation.
213                - self.last_velosity (float): The object's last recorded velocity.
214        """
215
216        return np.array([self.pos[0], self.pos[1], self.angle, self.last_velosity])
