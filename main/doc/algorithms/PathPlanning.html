180    def is_goal(self):
181        """
182        Checks if the ball is in the goal.
183        Returns:
184            bool: True if the ball is in the goal, False otherwise
185        """
186
187        if self.ball_pos is not None and self.goal_targer is not None:
188            # vec_behind = self.goal_targer - self.objects[DataClasses.BLUE][0][0:2]
189            # ver_beh_rot = self.mat_rot(-np.pi/2, vec_behind)
190            blue_vec = self.objects[DataClasses.BLUE][1][0:2] - self.objects[DataClasses.BLUE][0][0:2]
191            t = np.dot(self.robot_pos[:2] - self.objects[DataClasses.BLUE][0][0:2], blue_vec) / np.linalg.norm(blue_vec) ** 2
192            t = np.clip(t, 0, 1)
193            robot_dist = np.linalg.norm(self.robot_pos[:2] - (self.objects[DataClasses.BLUE][0][0:2] + t * blue_vec))
194            behind_goal = self.goal_targer - self.robot_pos[:2]
195            projection = (np.dot(blue_vec, behind_goal) / np.dot(blue_vec, blue_vec)) * blue_vec
196            behind_goal = behind_goal - projection
197            behind_goal = (behind_goal / np.linalg.norm(behind_goal) * PlanningParm.GOAL_CHECK)
198            dis_to_target = np.linalg.norm(self.ball_pos - (self.goal_targer - behind_goal))
199            dis_to_check = np.linalg.norm(self.ball_pos - (self.goal_targer + behind_goal))
200            if dis_to_target > dis_to_check or robot_dist < PlanningParm.ROBOT_DIST_MIN:
201                return True
202        return False

224    def generate_way_around(self, start, end, problem):
225        """
226        Generates a path around the problem point by finding the shortest path through the tangent points.
227        Args:
228            start (np.ndarray): Start point
229            end (np.ndarray): End point
230            problem (np.ndarray): Problem point
231        Returns:
232            np.ndarray: Path points for the alternative path
233        """
234
235        if np.linalg.norm(start - problem) < PlanningParm.CLEARANCE or np.linalg.norm(end - problem) < PlanningParm.CLEARANCE:
236            return np.array([start, end])
237
238        # Finds the tangent points from the start and end points to the problem point
239        tangent_point1, tangent_point2 = self.find_tangent_points(start, problem, PlanningParm.CLEARANCE)
240        tangent_point3, tangent_point4 = self.find_tangent_points(end, problem, PlanningParm.CLEARANCE)
241
242        # Creates a list of all possible combinations of tangent points
243        combinations = [(tangent_point1, tangent_point3), (tangent_point1, tangent_point4), (tangent_point2, tangent_point3), (tangent_point2, tangent_point4)]
244
245        shortest_inter, min_distance = None, float('inf')
246        for tangent_point_A, tangent_point_B in combinations:
247            intersection = self.find_intersection(start, tangent_point_A, end, tangent_point_B)  # finds the intersection point of the two tangent lines
248            distance = self.path_length([start, intersection, end])
249            if distance < min_distance:
250                min_distance = distance
251                shortest_inter = intersection
252
253        return np.array([start, shortest_inter, end])

313    def point_distance_from_line(self, P, A, B):
314        """
315        Calculates the perpendicular distance of a point from the line defined by points A and B.
316        Args:
317            P (np.ndarray): Point to check
318            A (np.ndarray): Start point of the line
319            B (np.ndarray): End point of the line
320        Returns:
321            float: Distance from the point to the line
322        """
323
324        AB = B - A
325        AP = P - A
326
327        # Projecton of AP onto AB
328        AB_dot_AB = np.dot(AB, AB)
329        AP_dot_AB = np.dot(AP, AB)
330
331        t = AP_dot_AB / AB_dot_AB
332
333        # If t is outside the segment, return the distance to the nearest endpoint
334        if t < 0.0:
335            nearest_point = A
336        elif t > 1.0:
337            nearest_point = B
338        else:
339            nearest_point = A + t * AB
340
341        return np.linalg.norm(P - nearest_point)

343    def find_tangent_points(self, start, center, radius):
344        """
345        Finds two tangent points from start point to circle.
346        Args:
347            start (np.ndarray): Start point
348            center (np.ndarray): Center of the object
349            radius (float): Radius of the object perimeter
350        Returns:
351            np.ndarray: First tangent point
352            np.ndarray: Second tangent point
353        """
354
355        d = start - center
356        d_norm = np.linalg.norm(d)
357        angle = np.arccos(radius / d_norm)
358        d_unit = d / d_norm
359
360        dir1 = self.mat_rot(angle, d_unit)
361        dir2 = self.mat_rot(-angle, d_unit)
362        tangent_point1 = center + radius * dir1
363        tangent_point2 = center + radius * dir2
364
365        return tangent_point1, tangent_point2

390    def find_intersection(self, lineA_a, lineA_b, lineB_a, lineB_b):
391        """
392        Finds the intersection point of two lines defined by two points each.
393        1. lineA: (lineA_a, lineA_b)
394        2. lineB: (lineB_a, lineB_b)
395        3. Returns the intersection point if exists, else None
396        Args:
397            lineA_a (np.ndarray): Start point of line A
398            lineA_b (np.ndarray): End point of line A
399            lineB_a (np.ndarray): Start point of line B
400            lineB_b (np.ndarray): End point of line B
401        Returns:
402            np.ndarray: Intersection point if exists, else None
403        """
404
405        lineA = lineA_b - lineA_a
406        lineB = lineB_b - lineB_a
407        M = np.column_stack([lineA, -lineB])
408        rhs = lineB_a - lineA_a
409
410        if np.linalg.det(M) == 0:
411            return None  # Lines are parallel and do not intersect
412
413        t, s = np.linalg.solve(M, rhs)
414        intersection = lineA_a + t * lineA
415        return intersection

314    def point_distance_from_line(self, P, A, B):
315        """
316        Calculates the perpendicular distance of a point from the line defined by points A and B.
317        Args:
318            P (np.ndarray): Point to check
319            A (np.ndarray): Start point of the line
320            B (np.ndarray): End point of the line
321        Returns:
322            float: Distance from the point to the line
323        """
324
325        AB = B - A
326        AP = P - A
327
328        # Projecton of AP onto AB
329        AB_dot_AB = np.dot(AB, AB)
330        AP_dot_AB = np.dot(AP, AB)
331
332        t = AP_dot_AB / AB_dot_AB
333
334        # If t is outside the segment, return the distance to the nearest endpoint
335        if t < 0.0:
336            nearest_point = A
337        elif t > 1.0:
338            nearest_point = B
339        else:
340            nearest_point = A + t * AB
341
342        return np.linalg.norm(P - nearest_point)

345        """
346        Finds two tangent points from start point to circle.
347        Args:
348            start (np.ndarray): Start point
349            center (np.ndarray): Center of the object
350            radius (float): Radius of the object perimeter
351        Returns:
352            np.ndarray: First tangent point
353            np.ndarray: Second tangent point
354        """

356        d = start - center

364        tangent_point2 = center + radius * dir2

391    def find_intersection(self, lineA_a, lineA_b, lineB_a, lineB_b):
392        """
393        Finds the intersection point of two lines defined by two points each.
394        1. lineA: (lineA_a, lineA_b)
395        2. lineB: (lineB_a, lineB_b)
396        3. Returns the intersection point if exists, else None
397        Args:
398            lineA_a (np.ndarray): Start point of line A
399            lineA_b (np.ndarray): End point of line A
400            lineB_a (np.ndarray): Start point of line B
401            lineB_b (np.ndarray): End point of line B
402        Returns:
403            np.ndarray: Intersection point if exists, else None
404        """
405
406        lineA = lineA_b - lineA_a
407        lineB = lineB_b - lineB_a
408        M = np.column_stack([lineA, -lineB])
409        rhs = lineB_a - lineA_a
410
411        if np.linalg.det(M) == 0:
412            return None  # Lines are parallel and do not intersect
413
414        t, s = np.linalg.solve(M, rhs)
415        intersection = lineA_a + t * lineA
416        return intersection

314    def point_distance_from_line(self, P, A, B):
315        """
316        Calculates the perpendicular distance of a point from the line defined by points A and B.
317        Args:
318            P (np.ndarray): Point to check
319            A (np.ndarray): Start point of the line
320            B (np.ndarray): End point of the line
321        Returns:
322            float: Distance from the point to the line
323        """
324
325        AB = B - A
326        AP = P - A
327
328        # Projecton of AP onto AB
329        AB_dot_AB = np.dot(AB, AB)
330        AP_dot_AB = np.dot(AP, AB)
331
332        t = AP_dot_AB / AB_dot_AB
333
334        # If t is outside the segment, return the distance to the nearest endpoint
335        if t < 0.0:

337        elif t > 1.0:
338            nearest_point = B
339        else:
340            nearest_point = A + t * AB
341
342        return np.linalg.norm(P - nearest_point)

392        """
393        Finds the intersection point of two lines defined by two points each.
394        1. lineA: (lineA_a, lineA_b)
395        2. lineB: (lineB_a, lineB_b)
396        3. Returns the intersection point if exists, else None
397        Args:
398            lineA_a (np.ndarray): Start point of line A
399            lineA_b (np.ndarray): End point of line A

401            lineB_b (np.ndarray): End point of line B
402        Returns:
403            np.ndarray: Intersection point if exists, else None
404        """
405
406        lineA = lineA_b - lineA_a
407        lineB = lineB_b - lineB_a

411        if np.linalg.det(M) == 0:
412            return None  # Lines are parallel and do not intersect
413
414        t, s = np.linalg.solve(M, rhs)
415        intersection = lineA_a + t * lineA
416        return intersection
