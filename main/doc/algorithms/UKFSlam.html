 35            Performs the update step of the UKF using camera detections.

 37            Associates perceived data with existing landmarks.

 42    def __init__(self, x_size: int, alpha: float, beta: float, kappa: float):
 43        self.x = np.zeros(x_size)
 44        self.P = np.eye(x_size) * 0.5
 45        self.data_cls = np.zeros((0))
 46        self.landmarks = np.zeros((0, 3))  # x, y, class
 47        self.alpha = alpha
 48        self.beta = beta
 49        self.kappa = kappa
 50        self.n = x_size
 51        # compute weights for sigma points
 52        self.lambda_ = self.alpha**2 * (self.n + self.kappa) - self.n
 53        # mean weights
 54        self.Wm = np.full(2 * self.n + 1, 1 / (2 * (self.n + self.lambda_)))
 55        self.Wm[0] = self.lambda_ / (self.n + self.lambda_)
 56        # covariance weights
 57        self.Wc = np.full(2 * self.n + 1, 1 / (2 * (self.n + self.lambda_)))
 58        self.Wc[0] = self.lambda_ / (self.n + self.lambda_) + (1 - self.alpha**2 + self.beta)
 59        self.landmark_contestants = np.zeros((0, 5))  # x, y, cls, seen, last_occurence

 61    def sigma_points(self, x, P):
 62        """
 63        Compute sigma points.
 64        Args:
 65            x (np.ndarray): mean
 66            P (np.ndarray): covariance
 67        """
 68        # square root of (self.n + self.lambda_)*P
 69        U = np.linalg.cholesky((self.n + self.lambda_) * P).T
 70        # sigma points
 71        sigmas = np.zeros((2 * self.n + 1, self.n))
 72        sigmas[0] = x.flatten()
 73        sigmas[1: self.n + 1] = x + U
 74        sigmas[self.n + 1:] = x - U
 75
 76        return sigmas

 78    def f(self, x, u):
 79        """
 80        State transition function.
 81        Args:
 82            x (np.ndarray): state
 83            u (np.ndarray): control input
 84        """
 85        x[:, :3] += u
 86        return x

 88    def predict(self, u, old_u):
 89        """
 90        Predict step
 91        Args:
 92            u (np.ndarray): odometry
 93            old_u (np.ndarray): previous odometry
 94        """
 95        # compute delta position and rotation
 96        delta_theta = np.tan(np.arctan(u[2] - old_u[2]))
 97        delta_pos = np.hstack((rotate_points(u[:2] - old_u[:2], -u[2]), delta_theta))
 98
 99        # compute sigma points
100        sigmas = self.sigma_points(self.x, self.P)
101
102        # pass sigma points through state transition function
103        R = np.array([[np.cos(self.x[2]), np.sin(self.x[2])], [-np.sin(self.x[2]), np.cos(self.x[2])]])
104        delta_pos[:2] = rotate_points(delta_pos[:2], self.x[2])
105        self.sigmas_f = self.f(sigmas, delta_pos)
106        # compute unscented transform
107        # mean
108        self.x = np.dot(self.Wm, self.sigmas_f)
109
110        # reziduals from mean
111        y = self.sigmas_f[1:] - self.x  # for n > 0
112        y0 = np.expand_dims(self.sigmas_f[0] - self.x, axis=0)  # for n = 0
113        # change in covariance matrix due to transformation
114        cov_difference = self.Wc[1] * y.T @ y + self.Wc[0] * y0.T @ y0
115        # rewrite covariance matrix
116        self.P = cov_difference
117        self.P = cov_difference.T
118        # add process noise
119        F = block_diag(R, 1, np.zeros((self.n - 3, self.n - 3)))
120        Qc = np.diag(np.hstack(([config["position_var"]] * 2, [config["rotation_var"]], np.zeros(self.n - 3))))
121        Q = F @ Qc @ F.T
122        self.P += Q

124    def update(self, z, h, R):
125        """
126        Update step.
127        Args:
128            z (np.ndarray): measurement
129            h (function): measurement function - returns in the same shape as z
130            R (np.ndarray): measurement noise covariance
131        """
132        # compute sigma points
133        # sigmas = self.sigmas_f # use sigma points from predict step
134        sigmas = self.sigma_points(self.x, self.P)
135
136        # pass sigma points through measurement function
137        sigmas_h = h(sigmas)
138
139        # compute mean and covariance of transformed sigma points
140        z_mean = np.dot(self.Wm, sigmas_h)
141
142        # compute measurement covariance
143        z_diff = sigmas_h[1:] - z_mean
144        z_diff_0 = np.expand_dims(sigmas_h[0] - z_mean, axis=0)
145        Pzz = self.Wc[1] * z_diff.T @ z_diff + self.Wc[0] * z_diff_0.T @ z_diff_0 + R
146
147        # compute cross covariance
148        x_diff = sigmas[1:] - self.x
149        x_diff_0 = np.expand_dims(sigmas[0] - self.x, axis=0)
150        Pxz = self.Wc[1] * x_diff.T @ z_diff + self.Wc[0] * x_diff_0.T @ z_diff_0
151
152        # compute Kalman gain
153        K = Pxz @ np.linalg.inv(Pzz)
154
155        # update state and covariance
156        self.x += K @ (z - z_mean)
157        self.P -= K @ Pzz @ K.T
158
159        return self.x, self.P

163        Perform data association.
164        Args:
165            percep_data (np.ndarray): perceived data
166        Returns:
167            tuple: A tuple containing:

171        # convert landmark to local coordinates
172        local_landmarks = np.hstack(
173            (global_to_local(self.x[3:].reshape(-1, 2).copy(), self.x[:3]), np.expand_dims(self.data_cls, axis=1))

176        diff_matrix = np.array([[1, 0, 0], [0, 1, 0], [0, 0, 1e6]])

181            # find closest cones

183            # find closest cones that are within the pairing distance
184            percep_data_mask = dist_mat[np.arange(0, percep_data.shape[0]), closest_cones] < config["pairing_distance"]

145        z_diff_0 = np.expand_dims(sigmas_h[0] - z_mean, axis=0)
